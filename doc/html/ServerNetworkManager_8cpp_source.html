75void ServerNetworkManager::onUpdate(float deltaTime) {
76 (void)deltaTime; // not used
77
78 if (!host_ || gameOver_) return;
79
80 ENetEvent event;
81 enet_host_flush(host_);
82
83 broadcastGameState(); //faudrait mettre un timer pour pas l'envoyer tout le temps
84
85 while (enet_host_service(host_, &event, 0) > 0) {
86 switch (event.type) {
87 case ENET_EVENT_TYPE_CONNECT:
88 handleNewConnection(event.peer);
89 break;
90
91 case ENET_EVENT_TYPE_RECEIVE: {
92 uint16_t packetType = event.packet->data[0];
93 const uint8_t* payload = event.packet->data + 1;
94 size_t payloadSize = event.packet->dataLength - 1;
95
96 if (packetType == PACKET_INITIAL_ACK) {
97 processInitialAck(event.peer);
98 } else if (packetType == PACKET_INPUT) {
99 uint16_t inputFlags = 0;
100 if (payloadSize >= sizeof(uint16_t)) {
101 std::memcpy(&inputFlags, payload, sizeof(uint16_t));
102 }
103 getRegistry().getSystem<NetworkReceiveSystem>()
104 .processInputCommand(inputFlags, event.peer);
105 }
106 enet_packet_destroy(event.packet);
107 break;
108 }
109
110 case ENET_EVENT_TYPE_DISCONNECT: {
111 std::cout << "Client disconnected: " << event.peer->connectID << std::endl;
112
113 forEach([&](ecs::Entity& entity) {
114 auto& netC = entity.getComponent<NetworkComponent>();
115 if (netC.peer == event.peer) {
116 std::string removedEntityUUID = entity.getName();
117
118 entity.remove();
119
120 broadcastEntityRemoval(removedEntityUUID);
121 }
122 });
123
124 connectedPeers_.erase(std::remove(connectedPeers_.begin(), connectedPeers_.end(), event.peer),
125 connectedPeers_.end());
126 peerInitialAck_.erase(event.peer);
127 initialEntitiesSent_.erase(event.peer);
128 enet_peer_reset(event.peer);
129 break;
130 }
131
132 default:
133 break;
134 }
135 }
136}
