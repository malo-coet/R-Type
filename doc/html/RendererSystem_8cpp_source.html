95void RendererSystem::updateAnimation(const RenderableComponent& renderable, float deltaTime) {
96 auto it = _textures.find(renderable.textureId);
97 if (it == _textures.end())
98 return;
99
100 RenderableInfos& textureInfo = it->second;
101
102 if (textureInfo.frameCount <= 1)
103 return;
104
105 textureInfo.animationTime += deltaTime;
106
107 if (textureInfo.animationTime >= textureInfo.frameDuration) {
108 textureInfo.currentFrame = (textureInfo.currentFrame + 1) % textureInfo.frameCount;
109 textureInfo.animationTime -= textureInfo.frameDuration;
110 }
111}

113void RendererSystem::drawEntity(float x, float y, const RenderableComponent& renderable, float deltaTime) {
114 auto it = _textures.find(renderable.textureId);
115 if (it == _textures.end()) {
116 std::cerr << "Erreur, pas de texture" << std::endl;
117 return;
118 }
119
120 RenderableInfos& textureInfo = it->second;
121
122 updateAnimation(renderable, deltaTime);
123
124 SDL_Rect src = {
125 textureInfo.currentFrame * textureInfo.frameWidth,
126 0,
127 textureInfo.frameWidth,
128 textureInfo.frameHeight
129 };
130
131 SDL_Rect dest = {
132 static_cast<int>(x),
133 static_cast<int>(y),
134 static_cast<int>(textureInfo.frameWidth * textureInfo.scale),
135 static_cast<int>(textureInfo.frameHeight * textureInfo.scale)
136 };
137
138 applyColorFilter(textureInfo.texture);
139
140 SDL_RenderCopy(renderer_, textureInfo.texture, &src, &dest);
141}

144// Load a texture
145SDL_Texture* RendererSystem::loadTexture(const std::string& path) {
146 SDL_RWops* rw = SDL_RWFromFile(path.c_str(), "rb");
147 if (!rw) {
148 std::cerr << "Failed to open file " << path << ": " << SDL_GetError() << std::endl;
149 return nullptr;
150 }
151
152 SDL_Surface* surface = IMG_LoadPNG_RW(rw);
153 SDL_RWclose(rw);
154
155 // if the surface is null, try to load it as a BMP
156 if (!surface) {
157 std::cerr << "Failed to load PNG " << path << ": " << IMG_GetError() << std::endl;
158 return nullptr;
159 }
160
161 // create a texture from the surface
162 SDL_Texture* texture = SDL_CreateTextureFromSurface(renderer_, surface);
163 SDL_FreeSurface(surface);
164
165 // if the texture is null, print an error
166 if (!texture) {
167 std::cerr << "Failed to create texture from " << path << ": " << SDL_GetError() << std::endl;
168 }
169
170 return texture;
171}

173void RendererSystem::cycleFilter() {
174 switch (currentFilter) {
175 case ColorFilter::NORMAL: currentFilter = ColorFilter::PROTANOPIA; break;
176 case ColorFilter::PROTANOPIA: currentFilter = ColorFilter::DEUTERANOPIA; break;
177 case ColorFilter::DEUTERANOPIA: currentFilter = ColorFilter::TRITANOPIA; break;
178 case ColorFilter::TRITANOPIA: currentFilter = ColorFilter::GRAYSCALE; break;
179 case ColorFilter::GRAYSCALE: currentFilter = ColorFilter::NORMAL; break;
180 }
181 std::cout << "Filter changed to " << static_cast<int>(currentFilter) << std::endl;
182}

184void RendererSystem::applyColorFilter(SDL_Texture* texture) {
185 switch (currentFilter) {
186 case ColorFilter::PROTANOPIA:
187 SDL_SetTextureColorMod(texture, 200, 255, 255);
188 break;
189 case ColorFilter::DEUTERANOPIA:
190 SDL_SetTextureColorMod(texture, 255, 200, 255);
191 break;
192 case ColorFilter::TRITANOPIA:
193 SDL_SetTextureColorMod(texture, 255, 255, 200);
194 break;
195 case ColorFilter::VIBRANT:
196 SDL_SetTextureColorMod(texture, 255, 180, 180);
197 break;
198 case ColorFilter::GRAYSCALE:
199 SDL_SetTextureColorMod(texture, 128, 128, 128);
200 break;
201 case ColorFilter::NORMAL:
202 default:
203 SDL_SetTextureColorMod(texture, 255, 255, 255);
204 break;
205 }
206}

208void RendererSystem::drawText(const std::string& text, int x, int y) {
209 SDL_Color color = {255, 255, 255, 255};
210 SDL_Surface* surface = TTF_RenderText_Solid(font_, text.c_str(), color);
211 if (!surface) {
212 std::cerr << "Failed to create text surface: " << TTF_GetError() << std::endl;
213 return;
214 }
215
216 SDL_Texture* texture = SDL_CreateTextureFromSurface(renderer_, surface);
217 SDL_FreeSurface(surface);
218
219 if (!texture) {
220 std::cerr << "Failed to create text texture: " << SDL_GetError() << std::endl;
221 return;
222 }
223
224 SDL_Rect dest = {x, y, surface->w, surface->h};
225 SDL_RenderCopy(renderer_, texture, nullptr, &dest);
226
227 SDL_DestroyTexture(texture);
228}
