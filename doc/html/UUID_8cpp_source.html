38 std::array<uint8_t, 16> uuidStringToByteArray(const std::string &uuidString) {
39 std::array<uint8_t, 16> uuidBytes{};
40 int index = 0;
41
42 for (char c : uuidString) {
43 if (c == '-')
44 continue;
45
46 uint8_t hex = 0;
47
48 if (c >= '0' && c <= '9')
49 hex = c - '0';
50 else if (c >= 'a' && c <= 'f')
51 hex = c - 'a' + 10;
52 else if (c >= 'A' && c <= 'F')
53 hex = c - 'A' + 10;
54
55 if (index % 2 == 0)
56 uuidBytes[index / 2] = hex << 4;
57 else
58 uuidBytes[index / 2] |= hex;
59 ++index;
60 }
61 return uuidBytes;
62 }
