122ecs::Entity EntityFactory::createEntityFromState(ecs::Registry& registry, const EntityState& state) {
123 ecs::Entity entity = registry.spawnEntity(uuid::byteArrayToUuidString(state.entityId));
124 entity.addComponent<position>({state.x, state.y});
125 entity.addComponent<velocity>({state.dx, state.dy});
126 entity.addComponent<typeComponent>({static_cast<EntityType>(state.entityType)});
127 entity.addComponent<RenderableComponent>({state.entityType});
128 switch (static_cast<EntityType>(state.entityType)) {
129 case EntityType::PLAYER :
130 entity.addComponent<Faction>({1});
131 entity.addComponent<Health>({10, 10});
132 entity.addComponent<Hitbox>({106, 42});
133 break;
134 case EntityType::PROJECTILE :
135 entity.addComponent<Faction>({1});
136 entity.addComponent<Health>({1, 1});
137 entity.addComponent<Hitbox>({51, 26});
138 entity.addComponent<AudioComponent>({AudioComponent::Type::SoundEffect, "laser", false, SDL_MIX_MAXVOLUME});
139 break;
140 case EntityType::ENEMY :
141 entity.addComponent<Faction>({2});
142 entity.addComponent<Health>({1, 1});
143 entity.addComponent<Hitbox>({69, 76});
144 break;
145 case EntityType::ENEMY_MED :
146 entity.addComponent<Faction>({2});
147 entity.addComponent<Health>({5, 5});
148 entity.addComponent<Hitbox>({386, 326});
149 break;
150 case EntityType::ENEMY_BIG :
151 entity.addComponent<Faction>({2});
152 entity.addComponent<Health>({10, 10});
153 entity.addComponent<Hitbox>({380, 488});
154 entity.addComponent<AudioComponent>({AudioComponent::Type::SoundEffect, "boss", false, SDL_MIX_MAXVOLUME});
155 break;
156 case EntityType::ENEMY_PROJECTILE :
157 entity.addComponent<Faction>({2});
158 entity.addComponent<Health>({1, 1});
159 entity.addComponent<Hitbox>({51, 51});
160 break;
161 case EntityType::ASTEROID :
162 entity.addComponent<Faction>({2});
163 entity.addComponent<Health>({1, 1});
164 entity.addComponent<Hitbox>({160, 135});
165 break;
166 default:
167 break;
168 }
169 return entity;
170}
