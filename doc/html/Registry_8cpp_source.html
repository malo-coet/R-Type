22ecs::Entity ecs::Registry::spawnEntity()
23{
24 Entity::Id id;
25
26 if (!_reusable_entity.empty()) {
27 id = _reusable_entity.top();
28 _reusable_entity.pop();
29 } else {
30 id = _next_entity++;
31 _entities.resize(id + 1);
32 _ComponentsFilter.resize(id + 1);
33 }
34
35 _entities[id].entity = Entity{id, *this};
36 _entities[id].isEnabled = true;
37 _entities[id].isValid = true;
38
39 enableEntity(_entities[id].entity);
40
41 return _entities[id].entity;

211void ecs::Registry::actionRemove(Entity::Id id)
212{
213 _systems.forEach([&](System &sytem){
214 sytem.detachEntity(_entities[id].entity);

216 _entities[id].isValid = false;

259void ecs::Registry::applyAction(EntityActions const &action)
260{
261 if (!isEntityValid(action.id))
262 throw std::runtime_error("Entity not valid");
263
264 switch (action.action) {
265 case EntityActions::Action::Enable:
266 actionEnable(action.id);
267 break;
268
269 case EntityActions::Action::Disable:
270 actionDisable(action.id);
271 break;
272
273 case EntityActions::Action::Refresh:
274 actionRefresh(action.id);
275 break;
276
277 case EntityActions::Action::Remove:
278 actionRemove(action.id);
279 break;
280 }
