43void ecs::System::detachEntity(Entity const &entity)
44{
45 EntityStatus const status = getEntityStatus(entity);
46
47 if (status != EntityStatus::NotAttached) {
48 if (status == EntityStatus::Enabled) {
49 _enableEntities.erase(std::remove(_enableEntities.begin(), _enableEntities.end(), entity));
