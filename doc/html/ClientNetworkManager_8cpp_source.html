58void ClientNetworkManager::onUpdate(float deltaTime)
59{
60 if (!host_) return;
61
62 ENetEvent event;
63 enet_host_flush(host_);
64
65 while (enet_host_service(host_, &event, 0) > 0) {
66 switch (event.type) {
67 case ENET_EVENT_TYPE_RECEIVE: {
68 uint8_t packetType = event.packet->data[0];
69 const uint8_t* payload = event.packet->data + 1;
70 size_t payloadSize = event.packet->dataLength - 1;
71
72 switch (packetType) {
73 case PACKET_PLAYER_ID:
74 processPlayerId(payload, payloadSize);
75 break;
76 case PACKET_INITIAL_ENTITIES:
77 processEntityUpdate(payload, payloadSize);
78 sendInitialAck();
79 break;
80 case PACKET_ENTITY_UPDATE:
81 processEntityUpdate(payload, payloadSize);
82 break;
83 case PACKET_ENTITY_REMOVE:
84 processEntityRemove(payload, payloadSize);
85 break;
86 case PACKET_GAME_STATE: {
87 if (payloadSize < sizeof(uint16_t) + sizeof(int)) {
88 std::cerr << "Paquet d'état de jeu trop petit\n";
89 break;
90 }
91 uint16_t wave;
92 int health;
93 std::memcpy(&wave, payload, sizeof(uint16_t));
94 std::memcpy(&health, payload + sizeof(uint16_t), sizeof(int));
95
96 auto maybeGlobal = getRegistry().getEntity("GlobalState");
97 if (maybeGlobal.has_value()) {
98 maybeGlobal.value().getComponent<GameStateComponent>().currentWave = wave;
99 maybeGlobal.value().getComponent<GameStateComponent>().playerHealth = health;
100 }
101 break;
102 }
103 default:
104 std::cerr << "Unknown packet type received: " << static_cast<int>(packetType) << "\n";
105 break;
106 }
107
108 enet_packet_destroy(event.packet);
109 break;
110 }
111
112 case ENET_EVENT_TYPE_DISCONNECT:
113 std::cout << "Disconnected from server.\n";
114 stopPingThread();
115 peer_ = nullptr;
116 break;
117
118 default:
119 break;
120 }
121 }
122}
