29void SpawnFromConfigSystem::onStart()
30{

32 std::ifstream file(rtypePath + "/RType/Server/waveConfig.json");
33 if (!file) {
34 throw std::runtime_error("Could not open file");
35 }
36
37 json jsonData;
38 file >> jsonData;
39
40 for (const auto& waveData : jsonData) {
41 Wave wave;
42 for (const auto& enemyData : waveData) {
43 Enemy enemy = {
44 enemyData.at("size"),
45 {enemyData.at("position")[0], enemyData.at("position")[1]},
46 enemyData.at("number")
47 };
48 wave.enemies.push_back(enemy);
49 }
50 waves_.push_back(wave);
51 }
52}

54void SpawnFromConfigSystem::spawnWave(const Wave& wave) {
55 for (const auto& enemy : wave.enemies) {
56 for (int i = 0; i < enemy.number; i++) {
57 if (enemy.size == 1)
58 EntityFactory::createEnemy(getRegistry(), enemy.position.first, enemy.position.second);

60 EntityFactory::createEnemyMed(getRegistry(), enemy.position.first, enemy.position.second);
61 if (enemy.size == 3)
62 EntityFactory::createEnemyBig(getRegistry(), enemy.position.first, enemy.position.second);
63 }
